jQuery Plugin: accelerometerSlider : Demo #5 - public plugin methods - prev, next, add and remove

the plugin leaves the dom tree untouched - until you add or delete something


prev | next | add image | add div | remove
$('#slider').accelerometerSlider( { effect: 'slidein', effectDuration: 350, responsive : true });

function addDiv()
{
    var div = document.createElement('div');
    $('#slider').data('accelerometerSlider').add(div);
}

function addImage()
{
    var img = document.createElement('img');
    img.src = 'http://placehold.it/350x194';

    $('#slider').data('accelerometerSlider').add(img);
}
